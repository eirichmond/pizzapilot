3. Use WordPress Settings API

```php
class PizzaPilot_Admin_Settings {
    public function __construct() {
        add_action('admin_menu', array($this, 'add_admin_menu'));
        add_action('admin_init', array($this, 'register_settings'));
    }
    
    public function add_admin_menu() {
        add_submenu_page(
            'woocommerce',
            'PizzaPilot Settings',
            'PizzaPilot',
            'manage_options',
            'pizzapilot-settings',
            array($this, 'settings_page')
        );
    }
    
    public function register_settings() {
        register_setting('pizzapilot_settings', 'pizzapilot_enabled', array('type' => 'boolean', 'default' => true));
        register_setting('pizzapilot_settings', 'pizzapilot_delivery_postcode', array('type' => 'string'));
        register_setting('pizzapilot_settings', 'pizzapilot_delivery_radius', array('type' => 'number', 'default' => 5));
        register_setting('pizzapilot_settings', 'pizzapilot_radius_unit', array('type' => 'string', 'default' => 'km'));
        
        add_settings_section('pizzapilot_general', 'General Settings', array($this, 'section_callback'), 'pizzapilot-settings');
        
        add_settings_field('pizzapilot_enabled', 'Enable PizzaPilot', array($this, 'enabled_callback'), 'pizzapilot-settings', 'pizzapilot_general');
        add_settings_field('pizzapilot_delivery_postcode', 'Delivery Postcode', array($this, 'delivery_postcode_callback'), 'pizzapilot-settings', 'pizzapilot_general');
        add_settings_field('pizzapilot_delivery_radius', 'Delivery Radius', array($this, 'delivery_radius_callback'), 'pizzapilot-settings', 'pizzapilot_general');
        add_settings_field('pizzapilot_same_day_only', 'Same-Day Delivery Only', array($this, 'same_day_only_callback'), 'pizzapilot-settings', 'pizzapilot_general');
    }
    
    // Implement callback methods for each setting field...
}
```
