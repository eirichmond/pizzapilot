class PizzaPilot_Slot_Manager {
    public function __construct() {
        add_action('woocommerce_order_status_changed', array($this, 'handle_order_status_change'), 10, 4);
        add_action('woocommerce_checkout_update_order_meta', array($this, 'update_slot_availability'));
    }
    
    public function count_pizzas_in_order($order_id) {
        $order = wc_get_order($order_id);
        $pizza_count = 0;
        
        foreach ($order->get_items() as $item) {
            // In free version, we count each product as 1 pizza
            $pizza_count += $item->get_quantity();
        }
        
        return $pizza_count;
    }
    
    public function update_slot_availability($order_id) {
        global $wpdb;
        $table_name = $wpdb->prefix . 'pizzapilot_order_slots';
        
        $slot_id = get_post_meta($order_id, '_pizzapilot_time_slot_id', true);
        if (!$slot_id) return;
        
        $pizza_count = $this->count_pizzas_in_order($order_id);
        
        // Update slot availability
        $wpdb->query($wpdb->prepare(
            "UPDATE $table_name 
            SET availability = availability - %d, 
                order_id = %d 
            WHERE slot_id = %d",
            $pizza_count,
            $order_id,
            $slot_id
        ));
    }
    
    public function handle_order_status_change($order_id, $from_status, $to_status, $order) {
        // If order is cancelled, refund the slot availability
        if ($to_status === 'cancelled') {
            $this->release_slot($order_id);
        }
    }

}
