class PizzaPilot_WooCommerce_Integration {
    public function __construct() {
        add_action('woocommerce_admin_order_data_after_shipping_address', array($this, 'display_slot_in_admin_order'));
        add_action('woocommerce_email_after_order_table', array($this, 'add_slot_to_emails'), 10, 4);
        add_action('woocommerce_order_details_after_order_table', array($this, 'add_slot_to_order_details'));
    }
    
    public function display_slot_in_admin_order($order) {
        $slot_id = get_post_meta($order->get_id(), '_pizzapilot_time_slot_id', true);
        if (!$slot_id) return;
        
        $slot_info = $this->get_slot_info($slot_id);
        if (!$slot_info) return;
        
        echo '<p><strong>Delivery Time Slot:</strong> ';
        echo date('F j, Y', strtotime($slot_info->date)) . ', ';
        echo date('g:i A', strtotime($slot_info->start_time)) . ' - ' . date('g:i A', strtotime($slot_info->end_time));
        echo '</p>';
    }
    
    public function add_slot_to_emails($order, $sent_to_admin, $plain_text, $email) {
        $slot_id = get_post_meta($order->get_id(), '_pizzapilot_time_slot_id', true);
        if (!$slot_id) return;
        
        $slot_info = $this->get_slot_info($slot_id);
        if (!$slot_info) return;
        
        if ($plain_text) {
            echo "\n\nDelivery Time Slot:\n";
            echo date('F j, Y', strtotime($slot_info->date)) . ', ';
            echo date('g:i A', strtotime($slot_info->start_time)) . ' - ' . date('g:i A', strtotime($slot_info->end_time));
        } else {
            echo '<h2>Delivery Time Slot</h2>';
            echo '<p>';
            echo date('F j, Y', strtotime($slot_info->date)) . ', ';
            echo date('g:i A', strtotime($slot_info->start_time)) . ' - ' . date('g:i A', strtotime($slot_info->end_time));
            echo '</p>';
        }
    }
    
    public function add_slot_to_order_details($order) {
        $slot_id = get_post_meta($order->get_id(), '_pizzapilot_time_slot_id', true);
        if (!$slot_id) return;
        
        $slot_info = $this->get_slot_info($slot_id);
        if (!$slot_info) return;
        
        echo '<section class="woocommerce-delivery-slot">';
        echo '<h2>Delivery Time Slot</h2>';
        echo '<p>';
        echo date('F j, Y', strtotime($slot_info->date)) . ', ';
        echo date('g:i A', strtotime($slot_info->start_time)) . ' - ' . date('g:i A', strtotime($slot_info->end_time));
        echo '</p>';
        echo '</section>';
    }
    
    private function get_slot_info($slot_id) {
        global $wpdb;
        $table_name = $wpdb->prefix . 'pizzapilot_order_slots';
        
        return $wpdb->get_row($wpdb->prepare(
            "SELECT * FROM $table_name WHERE slot_id = %d",
            $slot_id
        ));
    }
}
