class PizzaPilot_Assets {
    public function __construct() {
        add_action('wp_enqueue_scripts', array($this, 'enqueue_frontend_assets'));
        add_action('admin_enqueue_scripts', array($this, 'enqueue_admin_assets'));
    }
    
    public function enqueue_frontend_assets() {
        // Only enqueue on checkout and cart pages
        if (is_checkout() || is_cart()) {
            wp_enqueue_style(
                'pizzapilot-frontend',
                PIZZAPILOT_PLUGIN_URL . 'assets/css/frontend.css',
                array(),
                PIZZAPILOT_VERSION
            );
            
            wp_enqueue_script(
                'pizzapilot-frontend',
                PIZZAPILOT_PLUGIN_URL . 'assets/js/frontend.js',
                array('jquery'),
                PIZZAPILOT_VERSION,
                true
            );
            
            wp_localize_script('pizzapilot-frontend', 'pizzapilot_params', array(
                'ajax_url' => admin_url('admin-ajax.php'),
                'nonce' => wp_create_nonce('pizzapilot-nonce'),
            ));
        }
    }
    
    public function enqueue_admin_assets($hook) {
        // Only enqueue on plugin admin pages
        if (strpos($hook, 'pizzapilot') !== false) {
            wp_enqueue_style(
                'pizzapilot-admin',
                PIZZAPILOT_PLUGIN_URL . 'assets/css/admin.css',
                array(),
                PIZZAPILOT_VERSION
            );
            
            wp_enqueue_script(
                'pizzapilot-admin',
                PIZZAPILOT_PLUGIN_URL . 'assets/js/admin.js',
                array('jquery'),
                PIZZAPILOT_VERSION,
                true
            );
            
            wp_localize_script('pizzapilot-admin', 'pizzapilot_admin_params', array(
                'ajax_url' => admin_url('admin-ajax.php'),
                'nonce' => wp_create_nonce('pizzapilot-admin-nonce'),
            ));
        }
    }
}
