class PizzaPilot_i18n {
    public function __construct() {
        add_action('plugins_loaded', array($this, 'load_plugin_textdomain'));
    }
    
    public function load_plugin_textdomain() {
        load_plugin_textdomain(
            'pizzapilot',
            false,
            dirname(plugin_basename(__FILE__)) . '/languages/'
        );
    }
}

// Example of internationalized strings

// Date/time formatting with localization
function pizzapilot_format_time($time_string) {
    // Convert to timestamp
    $timestamp = strtotime($time_string);
    
    // Format according to WordPress settings
    return date_i18n(get_option('time_format'), $timestamp);
}

function pizzapilot_format_date($date_string) {
    // Convert to timestamp
    $timestamp = strtotime($date_string);
    
    // Format according to WordPress settings
    return date_i18n(get_option('date_format'), $timestamp);
}

// Generate POT file
// Use WP-CLI: wp i18n make-pot . languages/pizzapilot.pot
