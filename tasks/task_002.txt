# Description: Create the database table for storing time slots with the required fields as specified in the PRD.
# Details:
Create the `pizzapilot_order_slots` table with the following fields:
- slot_id (primary key, auto increment)
- date (date)
- start_time (time)
- end_time (time)
- availability (int, represents remaining pizza quantity)
- order_id (int, nullable, for tracking which order is using this slot)
- created_at (datetime)
- updated_at (datetime)

```php
public static function activate() {
    global $wpdb;
    $charset_collate = $wpdb->get_charset_collate();
    $table_name = $wpdb->prefix . 'pizzapilot_order_slots';
    
    $sql = "CREATE TABLE $table_name (
        slot_id bigint(20) NOT NULL AUTO_INCREMENT,
        date date NOT NULL,
        start_time time NOT NULL,
        end_time time NOT NULL,
        availability int(11) NOT NULL DEFAULT 0,
        order_id bigint(20) DEFAULT NULL,
        created_at datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
        updated_at datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
        PRIMARY KEY  (slot_id),
        KEY date (date),
        KEY order_id (order_id)
    ) $charset_collate;";
    
    require_once(ABSPATH . 'wp-admin/includes/upgrade.php');
    dbDelta($sql);
}
```

# Test Strategy:
