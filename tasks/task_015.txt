class PizzaPilot_Documentation {
    public function __construct() {
        add_action('admin_head', array($this, 'add_help_tabs'));
        add_action('admin_menu', array($this, 'add_help_page'));
        add_filter('plugin_action_links_' . plugin_basename(PIZZAPILOT_PLUGIN_DIR . 'pizzapilot.php'), array($this, 'add_plugin_links'));
    }
    
    public function add_help_tabs() {
        $screen = get_current_screen();
        
        // Only add help tabs to plugin pages
        if (!$screen || strpos($screen->id, 'pizzapilot') === false) {
            return;
        }
        
        // Settings page help
        if ($screen->id === 'woocommerce_page_pizzapilot-settings') {
            $screen->add_help_tab(array(
                'id' => 'pizzapilot-settings-overview',
                'title' => __('Overview', 'pizzapilot'),
                'content' => '<p>' . __('The settings page allows you to configure your delivery options and time slots.', 'pizzapilot') . '</p>'
            ));
            
            $screen->add_help_tab(array(
                'id' => 'pizzapilot-delivery-settings',
                'title' => __('Delivery Settings', 'pizzapilot'),
                'content' => '<p>' . __('Set your store location and maximum delivery radius. Customers outside this radius will not be able to select delivery.', 'pizzapilot') . '</p>'
            ));
        }
        
        // Time slots page help
        if ($screen->id === 'woocommerce_page_pizzapilot-time-slots') {
            $screen->add_help_tab(array(
                'id' => 'pizzapilot-slots-overview',
                'title' => __('Managing Time Slots', 'pizzapilot'),
                'content' => '<p>' . __('Create time slots for today to allow customers to select delivery times. Each slot has a maximum pizza capacity.', 'pizzapilot') . '</p>'
            ));
        }
        
        // Kitchen page help
        if ($screen->id === 'toplevel_page_pizzapilot-kitchen') {
            $screen->add_help_tab(array(
                'id' => 'pizzapilot-kitchen-overview',
                'title' => __('Kitchen View', 'pizzapilot'),
                'content' => '<p>' . __('This page shows orders grouped by time slot. You can mark orders as completed when they are ready for delivery.', 'pizzapilot') . '</p>'
            ));
        }
        
        // Add sidebar with links
        $screen->set_help_sidebar(
            '<p><strong>' . __('For more information:', 'pizzapilot') . '</strong></p>' .
            '<p><a href="https://example.com/pizzapilot-docs" target="_blank">' . __('Documentation', 'pizzapilot') . '</a></p>' .
            '<p><a href="https://example.com/pizzapilot-support" target="_blank">' . __('Support', 'pizzapilot') . '</a></p>'
        );
    }
    
    public function add_help_page() {
        add_submenu_page(
            'woocommerce',
            __('PizzaPilot Help', 'pizzapilot'),
            __('PizzaPilot Help', 'pizzapilot'),
            'manage_options',
            'pizzapilot-help',
            array($this, 'render_help_page')
        );
    }
    
    public function render_help_page() {
        echo '<div class="wrap">';
        echo '<h1>' . __('PizzaPilot Help & Documentation', 'pizzapilot') . '</h1>';
        
        echo '<div class="pizzapilot-help-section">';
        echo '<h2>' . __('Getting Started', 'pizzapilot') . '</h2>';
        echo '<ol>';
        echo '<li>' . __('Configure your delivery settings (postcode and radius)', 'pizzapilot') . '</li>';
        echo '<li>' . __('Create time slots for today', 'pizzapilot') . '</li>';
        echo '<li>' . __('Set maximum pizza capacity for each slot', 'pizzapilot') . '</li>';
        echo '<li>' . __('Monitor orders in the Kitchen View', 'pizzapilot') . '</li>';
        echo '</ol>';
        echo '</div>';
        
        echo '<div class="pizzapilot-help-section">';
        echo '<h2>' . __('Frequently Asked Questions', 'pizzapilot') . '</h2>';
        
        echo '<div class="pizzapilot-faq-item">';
        echo '<h3>' . __('Why can I only create slots for today?', 'pizzapilot') . '</h3>';
        echo '<p>' . __('The free version of PizzaPilot only supports same-day delivery. Upgrade to Pro for advanced scheduling features.', 'pizzapilot') . '</p>';
        echo '</div>';
        
        echo '<div class="pizzapilot-faq-item">';
        echo '<h3>' . __('How is the delivery radius calculated?', 'pizzapilot') . '</h3>';
        echo '<p>' . __('The free version uses a simple distance calculation based on postcodes. For more accurate delivery zones with mapping, upgrade to Pro.', 'pizzapilot') . '</p>';
        echo '</div>';
        
        // Add more FAQs...
        
        echo '</div>';
        
        echo '</div>'; // Close wrap
    }
    
    public function add_plugin_links($links) {
        $plugin_links = array(
            '<a href="' . admin_url('admin.php?page=pizzapilot-settings') . '">' . __('Settings', 'pizzapilot') . '</a>',
            '<a href="' . admin_url('admin.php?page=pizzapilot-help') . '">' . __('Help', 'pizzapilot') . '</a>',
        );
        
        return array_merge($plugin_links, $links);
    }
}
