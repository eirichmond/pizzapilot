class PizzaPilot_Order_Admin {
    public function __construct() {
        add_filter('manage_edit-shop_order_columns', array($this, 'add_order_slot_column'));
        add_action('manage_shop_order_posts_custom_column', array($this, 'display_order_slot_column_content'));
        add_action('restrict_manage_posts', array($this, 'add_time_slot_filter'));
        add_filter('request', array($this, 'filter_orders_by_slot'));
    }
    
    public function add_order_slot_column($columns) {
        $new_columns = array();
        
        foreach ($columns as $column_name => $column_info) {
            $new_columns[$column_name] = $column_info;
            
            if ($column_name === 'order_status') {
                $new_columns['order_slot'] = 'Delivery Slot';
            }
        }
        
        return $new_columns;
    }
    
    public function display_order_slot_column_content($column) {
        global $post;
        
        if ($column === 'order_slot') {
            $order = wc_get_order($post->ID);
            $slot_id = get_post_meta($order->get_id(), '_pizzapilot_time_slot_id', true);
            
            if ($slot_id) {
                $slot_info = $this->get_slot_info($slot_id);
                
                if ($slot_info) {
                    echo date('g:i A', strtotime($slot_info->start_time)) . ' - ' . date('g:i A', strtotime($slot_info->end_time));
                } else {
                    echo '—';
                }
            } else {
                echo '—';
            }
        }
    }
    
    public function add_time_slot_filter() {
        global $typenow, $wpdb;
        
        if ($typenow !== 'shop_order') {
            return;
        }
        
        $table_name = $wpdb->prefix . 'pizzapilot_order_slots';
        $today = date('Y-m-d');
        
        // Get today's slots
        $slots = $wpdb->get_results($wpdb->prepare(
            "SELECT slot_id, start_time, end_time 
            FROM $table_name 
            WHERE date = %s 
            ORDER BY start_time ASC",
            $today
        ));
        
        if (empty($slots)) {
            return;
        }
        
        $current_slot = isset($_GET['pizzapilot_slot']) ? absint($_GET['pizzapilot_slot']) : 0;
        
        echo '<select name="pizzapilot_slot">';
        echo '<option value="0">All time slots</option>';
        
        foreach ($slots as $slot) {
            $start = date('g:i A', strtotime($slot->start_time));
            $end = date('g:i A', strtotime($slot->end_time));
            echo '<option value="' . $slot->slot_id . '" ' . selected($current_slot, $slot->slot_id, false) . '>';
            echo $start . ' - ' . $end;
            echo '</option>';
        }
        
        echo '</select>';
    }
    
    public function filter_orders_by_slot($vars) {
        global $typenow, $wp_query;
        
        if ($typenow === 'shop_order' && isset($_GET['pizzapilot_slot']) && $_GET['pizzapilot_slot'] > 0) {
            $vars['meta_key'] = '_pizzapilot_time_slot_id';
            $vars['meta_value'] = absint($_GET['pizzapilot_slot']);
        }
        
        return $vars;
    }
    
    private function get_slot_info($slot_id) {
        global $wpdb;
        $table_name = $wpdb->prefix . 'pizzapilot_order_slots';
        
        return $wpdb->get_row($wpdb->prepare(
            "SELECT * FROM $table_name WHERE slot_id = %d",
            $slot_id
        ));
    }
}
